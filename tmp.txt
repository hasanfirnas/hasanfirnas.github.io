import openpyxl

def parse_excel_for_data(filename):
    workbook = openpyxl.load_workbook(filename)
    worksheet = workbook.active

    data_list = []
    for row in worksheet.iter_rows(min_row=2):  # Skip the header row
        ciid = row[1].value
        application_name = row[2].value
        ssoid_str = row[3].value
        ssoid = ssoid_str.split()[0]
        data_list.append([ciid, application_name, ssoid])

    return data_list

# Example usage:
filename = "your_excel_file.xlsx"
data_list = parse_excel_for_data(filename)

print(data_list)
