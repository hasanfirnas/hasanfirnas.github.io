import openpyxl

# Function to open the Excel file and print sheet names
def print_sheet_names(file_path):
    try:
        # Load the workbook
        workbook = openpyxl.load_workbook(file_path)
        
        # Print all sheet names
        print("Sheet names in the workbook:")
        for sheet in workbook.sheetnames:
            print(sheet)
    except FileNotFoundError:
        print(f"File not found: {file_path}")
    except Exception as e:
        print(f"An error occurred: {e}")

# Example usage
file_path = "your_file.xlsx"  # Replace with the path to your Excel file
print_sheet_names(file_path)
