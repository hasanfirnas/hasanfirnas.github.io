Sub SearchAndMark()
    Dim ws As Worksheet
    Dim searchText As String
    Dim searchArray() As String
    Dim lastRow As Long
    Dim i As Long, j As Long

    ' Define the worksheet and search text
    Set ws = ThisWorkbook.Sheets(1) ' Change to your sheet name if needed
    searchText = InputBox("Enter the text to search for, separated by commas:")
    searchArray = Split(searchText, ",") ' Split the text into an array

    ' Find the last row in Column C
    lastRow = ws.Cells(ws.Rows.Count, "C").End(xlUp).Row

    ' Loop through each row in Column C
    For i = 1 To lastRow
        Dim found As Boolean
        found = False

        ' Check each search term in the current row
        For j = LBound(searchArray) To UBound(searchArray)
            If InStr(1, ws.Cells(i, "C").Value, Trim(searchArray(j)), vbTextCompare) > 0 Then
                found = True
                Exit For
            End If
        Next j

        ' Write "Yes" in Column E if any search term is found
        If found Then
            ws.Cells(i, "E").Value = "Yes"
        Else
            ws.Cells(i, "E").Value = ""
        End If
    Next i

    MsgBox "Search complete. Results written in Column E.", vbInformation
End Sub
