function main(workbook: ExcelScript.Workbook) {
  const sheet = workbook.getActiveWorksheet();

  // List of column headers you want to KEEP
  const columnsToKeep = ["uid", "nameFull", "jobTitle", "assignmentStatus", "mailAddress"];

  // Get the first row (header row)
  const headerRow = sheet.getRange("1:1").getValues()[0];

  // Loop through header row from end to start to avoid index shifting when deleting
  for (let col = headerRow.length - 1; col >= 0; col--) {
    const header = headerRow[col];

    // If header is NOT in the list to keep, delete the column
    if (!columnsToKeep.includes(header as string)) {
      sheet.getRangeByIndexes(0, col, sheet.getUsedRange().getRowCount(), 1).delete(ExcelScript.DeleteShiftDirection.left);
    }
  }
}
