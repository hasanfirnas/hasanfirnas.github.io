Sub KeepOnlySpecificColumns()
    Dim ws As Worksheet
    Dim cell As Range
    Dim i As Long
    Dim columnsToKeep As Variant

    columnsToKeep = Array("uid", "nameFull", "jobTitle", "assignmentStatus", "mailAddress")
    Set ws = ThisWorkbook.Sheets(1) ' Adjust if your target sheet is not the first one

    ' Loop from last to first column
    For i = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column To 1 Step -1
        If IsError(Application.Match(ws.Cells(1, i).Value, columnsToKeep, 0)) Then
            ws.Columns(i).Delete
        End If
    Next i
End Sub
