function main(workbook: ExcelScript.Workbook) {
  // Get the active worksheet
  let sheet = workbook.getActiveWorksheet();

  // Get the data from Column E
  let columnE = sheet.getRange("E2:E" + sheet.getUsedRange().getLastRow()).getValues();

  // Prepare an array to store extracted DB instance IDs
  let extractedValues: string[][] = [];

  // Loop through each cell in Column E
  for (let i = 0; i < columnE.length; i++) {
    let cellValue = columnE[i][0]?.toString() || ""; // Get the cell value as a string

    // Extract the DB instance ID using regex
    let match = cellValue.match(/DB instance:\s*(.*?)<\/li>/);
    let extractedValue = match ? match[1].trim() : ""; // Get the text or leave it empty if no match

    // Add the extracted value to the output array
    extractedValues.push([extractedValue]);
  }

  // Write the extracted values to Column F
  sheet.getRange("F2:F" + (extractedValues.length + 1)).setValues(extractedValues);
}
