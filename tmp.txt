function main(workbook: ExcelScript.Workbook, searchStrings: string) {
  // Split the input string into an array of strings to search for
  let stringsToSearch = searchStrings.split(",").map(s => s.trim());

  // Get the active worksheet
  let sheet = workbook.getActiveWorksheet();

  // Get the used range of the worksheet
  let range = sheet.getUsedRange();

  // If no range is found, exit the script
  if (!range) {
    console.log("No data found in the worksheet.");
    return;
  }

  // Get the total number of rows in the used range
  let rowCount = range.getRowCount();

  // Initialize arrays to track found and not-found strings
  let foundStrings: string[] = [];
  let notFoundStrings: string[] = [...stringsToSearch];

  // Loop through each row to find the strings
  for (let i = 0; i < rowCount; i++) {
    let row = range.getRow(i);

    // Check if any cell in the row contains any of the search strings
    let cells = row.getValues()[0];
    stringsToSearch.forEach(searchString => {
      if (cells.some(cell => cell && cell.toString().includes(searchString))) {
        // If found, add to the foundStrings array (if not already added)
        if (!foundStrings.includes(searchString)) {
          foundStrings.push(searchString);
        }

        // Remove the string from the notFoundStrings array
        notFoundStrings = notFoundStrings.filter(s => s !== searchString);

        // Apply strikethrough formatting to the row
        let cellCount = row.getColumnCount();
        for (let j = 0; j < cellCount; j++) {
          let cell = row.getCell(0, j);
          cell.getFormat().getFont().setStrikethrough(true); // Apply strikethrough
          cell.getFormat().getFill().setColor("yellow");     // Apply yellow highlight
        }
      }
    });
  }

  // Create a new sheet for the summary
  let summarySheet = workbook.addWorksheet("Search Summary");

  // Add headings
  summarySheet.getRange("A1").setValue("Found Strings");
  summarySheet.getRange("B1").setValue("Not Found Strings");

  // Populate found strings in column A
  if (foundStrings.length > 0) {
    summarySheet.getRange(`A2:A${foundStrings.length + 1}`).setValues(foundStrings.map(s => [s]));
  }

  // Populate not-found strings in column B
  if (notFoundStrings.length > 0) {
    summarySheet.getRange(`B2:B${notFoundStrings.length + 1}`).setValues(notFoundStrings.map(s => [s]));
  }

  console.log("Search summary has been logged in the sheet: 'Search Summary'.");
}
