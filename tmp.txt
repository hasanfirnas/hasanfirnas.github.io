function main(workbook: ExcelScript.Workbook, searchStrings: string[]) {
    let sheet = workbook.getActiveWorksheet(); // Get the active worksheet.
    let range = sheet.getUsedRange(); // Get all data in the worksheet.
    let rowCount = range.getRowCount(); // Get the number of rows.
    let columnCount = range.getColumnCount(); // Get the number of columns.

    // To store strings not found
    let notFoundStrings: string[] = [];

    // Loop through each string in the searchStrings array
    for (let searchString of searchStrings) {
        let found = false;

        // Loop through each cell in the used range
        for (let row = 0; row < rowCount; row++) {
            for (let col = 0; col < columnCount; col++) {
                let cellValue = range.getCell(row, col).getText();

                // If a match is found
                if (cellValue === searchString) {
                    found = true;
                    let cell = range.getCell(row, col);
                    cell.getFormat().setHorizontalAlignment("Left");
                    cell.getFormat().getFont().setStrikethrough(true); // Apply strike-through.
                }
            }
        }

        // If the string was not found in the worksheet
        if (!found) {
            notFoundStrings.push(searchString);
        }
    }

    // Log or return the strings that were not found
    if (notFoundStrings.length > 0) {
        console.log("The following strings were not found: " + notFoundStrings.join(", "));
    }
}
