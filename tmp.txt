import pandas as pd
from openpyxl import load_workbook

# Load the Excel file with openpyxl
excel_path = 'your_excel_file.xlsx'
workbook = load_workbook(excel_path)
sheet = workbook.active

# Load the list of numbers
with open('numbers.txt') as f:
    numbers = set(map(int, f.readlines()))

# Start from the second row (index 2 in Excel, which is 1 in zero-based indexing)
for row in range(2, sheet.max_row + 1):
    cell_value = sheet[f'A{row}'].value
    if cell_value in numbers:
        sheet[f'L{row}'] = 'remove this'

# Save the workbook with changes
workbook.save('updated_excel_file.xlsx')
