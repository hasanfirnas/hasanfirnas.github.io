function main(workbook: ExcelScript.Workbook) {
  // Get the active worksheet
  const sheet = workbook.getActiveWorksheet();

  // Define the column for search (Column C) and the result column (Column E)
  const searchColumn = sheet.getRange("C:C").getValues();
  const resultColumn = sheet.getRange("E:E");

  // Prompt the user for search terms
  const searchTermsInput = prompt("Enter the text to search for, separated by commas:");
  if (!searchTermsInput) {
    console.log("No search terms provided. Exiting script.");
    return;
  }

  // Split the search terms into an array
  const searchTerms = searchTermsInput.split(",").map(term => term.trim());

  // Loop through each row and search for matches in Column C
  const resultValues = searchColumn.map(row => {
    const cellValue = row[0]?.toString().toLowerCase() || "";
    const isMatch = searchTerms.some(term => cellValue.includes(term.toLowerCase()));
    return [isMatch ? "Yes" : ""]; // Return "Yes" if match, otherwise blank
  });

  // Write the results to Column E
  resultColumn.setValues(resultValues);

  console.log("Script execution completed. Results written in Column E.");
}
